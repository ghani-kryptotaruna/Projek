Jika pakai sumber internetnya dari wifi maka settingan netplannya seperti dibawah ini
contoh: 

$ sudo vim /etc/netplan/50-cloud-init.yaml 
network:
  version: 2
  ethernets:
    enp0s3:
      dhcp4: no
      addresses:
        - 192.168.5.253/24
      gateway4: 192.168.5.1
      nameservers:
        addresses: [8.8.8.8, 1.1.1.1]

$ sudo netplan apply

$ sudo systemctl restart systemd-resolved

$ sudo vim /etc/resolv.conf           # tambahkan dipaling bawah line 25
nameserver 8.8.8.8

$ sudo apt update

Jika sumber internetnya dari kabel lan maka settingan netplan nya seperti dibawah ini
contoh:

$ sudo vim /etc/netplan/50-cloud-init.yaml
network:
  version: 2
  ethernets:
    enp0s3:
      dhcp4: no
      addresses:
        - 192.168.4.191/24
      routes:
        - to: default
          via: 192.168.4.1
      nameservers:
        addresses:
          - 192.168.4.1
          - 8.8.8.8

$ sudo netplan apply
